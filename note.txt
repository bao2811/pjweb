# 1. Backend - Clear cache
cd backend
php artisan config:clear
php artisan cache:clear

# 2. Stop tất cả terminal (Ctrl+C)

# 3. Start lại:
# Terminal 1
php artisan reverb:start

# Terminal 2
php artisan queue:work

# Terminal 3
php artisan serve

# 4. Frontend - Restart
cd ../frontend
# Ctrl+C để stop
npm run dev
